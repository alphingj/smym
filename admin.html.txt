        function addProduct() {
            const productName = document.getElementById('productName').value;
            const productPrice = parseFloat(document.getElementById('productPrice').value);
            if (productName && !isNaN(productPrice) && productPrice > 0) {
                const newProduct = document.createElement('div');
                newProduct.innerHTML = `<span>${productName} - â‚¹${productPrice}</span> <button onclick="addToCart('${productName}', ${productPrice})">Add to Cart</button>`;
                document.getElementById('productList').appendChild(newProduct);
                alert(`Product ${productName} added successfully!`);
                document.getElementById('productName').value = '';
                document.getElementById('productPrice').value = '';
            } else {
                alert('Please enter valid product name and price.');
            }
        }
    </script>
